    sudo hostnamectl set-hostname wokanhao.com
    sudo vim /etc/hosts
    update/append:
      127.0.0.1        localhost
      192.168.184.128  wokanhao.com

      nginx ->  server -> server_name -> 192.168.184.128
      nginx ->  server -> location -> proxy_pass -> 192.168.184.128:8080
